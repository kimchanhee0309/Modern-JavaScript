# (3) 배열 요소 접근
* **인덱스는 0부터 시작**
```Javascript
let fruits = ['사과', '바나나', '오렌지', '포도'];

// 인덱스로 접근
console.log(fruits[0]); // 사과
console.log(fruits[1]); // 바나나
console.log(fruits[2]); // 오렌지
console.log(fruits[3]); // 포도

// 마지막 요소
let lastIndex = fruits.length - 1;
console.log(fruits[lastIndex]); // 포도

// 없는 인덱스
console.log(fruits[10]); // undefined
```

# (4) 배열 요소 수정
```Javascript
let numbers = [1, 2, 3, 4, 5];

// 요소 변경
numbers[0] = 10;
console.log(numbers); // [10, 2, 3, 4, 5];

numbers[2] = 100;
console.log(numbers); // [10, 2, 100, 4, 5];
```

# (6) 배열 요소 추가/삭제★
* **끝에 추가/삭제(push/pop)**
```Javascript
let fruits = ['사과', '바나나'];

// push - 끝에 추가
fruits.push('오렌지');
console.log(fruits); // ['사과', '바나나', '오렌지']

fruits.push('포도', '딸기');
console.log(fruits); // ['사과', '바나나', '오렌지', '포도', '딸기']

// pop - 끝에서 제거
let last = fruit.pop();
console.log(last); // 딸기
console.log(fruits); // ['사과', '바나나', '오렌지', '포도']
```

* **앞에 추가/삭제(unshift/shift)**
```Javascript
let fruits = ['바나나', '오렌지'];

// unshift - 앞에 추가
fruits.unshift('사과');
console.log(fruits); // ['사과', '바나나', '오렌지']

// shift - 앞에서 제거
let first = fruits.shift();
console.log(first); // 사과
console.log(fruits); // ['바나나', '오렌지']

```

# (7) 배열 순회★★
* **for문**
```Javascript
let fruits = ['사과', '바나나', '오렌지'];

for (let i = 0; i < fruits.length; i++) {
  console.log(fruits[i]);
}
// 사과
// 바나나
// 오렌지
```

* **for..of문(ES6)**
```Javascript
let fruits = ['사과', '바나나', '오렌지'];

for (let fruit of fruits) {
  console.log(fruit);
}
// 사과
// 바나나
// 오렌지
```

# (8) 다차원 배열
```Javascript

let matrix = [
  [1, 2, 3],
  [4, 5, 6],
  [7, 8, 9]
];

console.log(matrix[0][0]); // 1
console.log(matrix[1][1]); // 5
console.log(matrix[2][2]); // 9

// 학생 성적 관리

  ['홍길동', 90, 85, 88],
  ['김철수', 78, 92, 81],
  ['이영희', 95, 89, 93]
];

console.log(students[0][1]); // 90
```
