# 1. 함수 선언과 표현식⭐️

<details>
  <summary>(1) 함수 선언문(Function Declaration)</summary>

  * **기본 형태**
  ```JavaScript
  function 함수명(매개변수) {
      // 실행할 코드
      return 반환값;
  }
  ```

  * **예제**
  ```JavaScript
  // 간단한 함수
  function greet() {
      console.log('안녕하세요!');
  }

  greet();  // 안녕하세요!

  // 매개변수가 있는 함수
  function sayHello(name) {
      console.log('안녕하세요, ' + name + '님!');
  }

  sayHello('홍길동');  // 안녕하세요, 홍길동님!

  // 반환값이 있는 함수
  function add(a, b) {
      return a + b;
  }

  let result = add(10, 20);
  console.log(result);  // 30
  ```

  * **특정**
  ```JavaScript
  // 1. 호이스팅됨 (선언 전 호출 가능)
  greet();  // 안녕하세요! (에러 없음)

  function greet() {
      console.log('안녕하세요!');
  }

  // 2. 함수명 필수
  // function (name) { } // SyntaxError
  ```
</details>

<details>
  <summary>(2) 함수 표현식(Function Expression)</summary>

  * **기본 형태**
  ```JavaScript
  const 변수명 = function(매개변수) {
      // 실행할 코드
      return 반환값;
  };
  ```

  * **예제**
  ```JavaScript
  // 익명 함수 표현식
  const greet = function() {
      console.log('안녕하세요!');
  };

  greet();  // 안녕하세요!

  // 매개변수와 반환값
  const add = function(a, b) {
      return a + b;
  };

  console.log(add(10, 20));  // 30
  ```

  * **특징**
  ```JavaScript
  // 1. 호이스팅 안 됨
  greet();   // TypeError: greet is not a function

  const greet = function() {
      console.log('안녕하세요!');
  };

  // 2. 변수에 할당되므로 재할당 가능 (let 사용 시)
  let myFunc = function() {
      console.log('첫 번째');
  };

  myFunc();  // 첫 번째

  myFunc = function() {
      console.log('두 번째');
  };

  myFunc();  // 두 번째
  ```
</details>

<details>
  <summary>(3) 함수 선언문 vs 함수 표현식</summary>

  |**특징**|**함수 선언문**|**함수 표현식**|
  |-----|-----|-----|
  |**형태**|`function name() {}`|`const name = functiuon() {}`|
  |**호이스팅**|O|X|
  |**이름**|필수|선택|
  |**재할당**|X|O(let 사용 시)|
  |**사용 시점**|선언 전에도 가능|선언 후에만 가능|

  * **언제 무엇을 사용할까?**
  ```JavaScript
  // 함수 선언문: 유틸리티 함수, 헬퍼 함수
  function formatDate(date) {
      //...
  }

  function validateEmail(email) {
      //...
  }

  // 함수 표현식: 변수에 할당, 조건부 생성
  const getUserData = function() {
      //...
  };

  // 조건에 따라 다른 함수 할당
  let operation;
  if (mode === 'add') {
      operation = function(a, b) { return a + b; };
  } else {
      operation = function(a, b) { return a - b; };
  }
  ```
</details>

<details>
  <summary>(4) return 문</summary>

  * **반환값**
  ```JavaScript
  function add(a, b) {
      return a + b;
  }

  let result = add(10, 20);
  console.log(result);  // 30
  ```

  * **return은 함수 종료**
  ```JavaScript
  function checkAge(age) {
      if (age < 18) {
          return '미성년자';
          console.log('이 코드는 실행 안 됨');
      }
      return '성인';
  }

  console.log(checkAge(15));  // 미성년자
  console.log(checkAge(20));  // 성인
  ```

  * **return 없으면 undefined**
  ```JavaScript
  function greet() {
      console.log('안녕하세요!');
      // return 없음
  }

  let result = greet();
  console.log(result);  // undefined
  ```

  * **조기 반환(Early Return) 패턴**
  ```JavaScript
  // 나쁜 예
  function processUser(user) {
      if (user) {
          if (user.age >= 18) {
              if (user.verified) {
                  // 실제 로직
                  return '처리 완료';
              } else {
                  return '인증 필요';
              }
          } else {
              return '미성년자';
          }
      } else {
          return '사용자 없음';
      }
  }

  // 좋은 예(조기 반환)
  function processUser(user) {
      if (!user) return '사용자 없음';
      if (user.age < 18) return '미성년자';
      if (!user.verified) return '인증 필요';

      // 실제 로직
      return '처리 완료';
  }
  ```
</details>
