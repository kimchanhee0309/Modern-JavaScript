# 1. 함수 선언과 표현식⭐️

<details>
  <summary>(1) 함수 선언문(Function Declaration)</summary>

  * **기본 형태**
  ```JavaScript
  function 함수명(매개변수) {
      // 실행할 코드
      return 반환값;
  }
  ```

  * **예제**
  ```JavaScript
  // 간단한 함수
  function greet() {
      console.log('안녕하세요!');
  }

  greet();  // 안녕하세요!

  // 매개변수가 있는 함수
  function sayHello(name) {
      console.log('안녕하세요, ' + name + '님!');
  }

  sayHello('홍길동');  // 안녕하세요, 홍길동님!

  // 반환값이 있는 함수
  function add(a, b) {
      return a + b;
  }

  let result = add(10, 20);
  console.log(result);  // 30
  ```

  * **특정**
  ```JavaScript
  // 1. 호이스팅됨 (선언 전 호출 가능)
  greet();  // 안녕하세요! (에러 없음)

  function greet() {
      console.log('안녕하세요!');
  }

  // 2. 함수명 필수
  // function (name) { } // SyntaxError
  ```
</details>

<details>
  <summary>(2) 함수 표현식(Function Expression)</summary>

  * **기본 형태**
  ```JavaScript
  const 변수명 = function(매개변수) {
      // 실행할 코드
      return 반환값;
  };
  ```

  * **예제**
  ```JavaScript
  // 익명 함수 표현식
  const greet = function() {
      console.log('안녕하세요!');
  };

  greet();  // 안녕하세요!

  // 매개변수와 반환값
  const add = function(a, b) {
      return a + b;
  };

  console.log(add(10, 20));  // 30
  ```

  * **특징**
  ```JavaScript
  // 1. 호이스팅 안 됨
  greet();   // TypeError: greet is not a function

  const greet = function() {
      console.log('안녕하세요!');
  };

  // 2. 변수에 할당되므로 재할당 가능 (let 사용 시)
  let myFunc = function() {
      console.log('첫 번째');
  };

  myFunc();  // 첫 번째

  myFunc = function() {
      console.log('두 번째');
  };

  myFunc();  // 두 번째
  ```
</details>

<details>
  <summary>(3) 함수 선언문 vs 함수 표현식</summary>

  |**특징**|**함수 선언문**|**함수 표현식**|
  |-----|-----|-----|
  |**형태**|`function name() {}`|`const name = functiuon() {}`|
  |**호이스팅**|O|X|
  |**이름**|필수|선택|
  |**재할당**|X|O(let 사용 시)|
  |**사용 시점**|선언 전에도 가능|선언 후에만 가능|

  * **언제 무엇을 사용할까?**
  ```JavaScript
  // 함수 선언문: 유틸리티 함수, 헬퍼 함수
  function formatDate(date) {
      //...
  }

  function validateEmail(email) {
      //...
  }

  // 함수 표현식: 변수에 할당, 조건부 생성
  const getUserData = function() {
      //...
  };

  // 조건에 따라 다른 함수 할당
  let operation;
  if (mode === 'add') {
      operation = function(a, b) { return a + b; };
  } else {
      operation = function(a, b) { return a - b; };
  }
  ```
</details>

<details>
  <summary>(4) return 문</summary>

  * **반환값**
  ```JavaScript
  function add(a, b) {
      return a + b;
  }

  let result = add(10, 20);
  console.log(result);  // 30
  ```

  * **return은 함수 종료**
  ```JavaScript
  function checkAge(age) {
      if (age < 18) {
          return '미성년자';
          console.log('이 코드는 실행 안 됨');
      }
      return '성인';
  }

  console.log(checkAge(15));  // 미성년자
  console.log(checkAge(20));  // 성인
  ```

  * **return 없으면 undefined**
  ```JavaScript
  function greet() {
      console.log('안녕하세요!');
      // return 없음
  }

  let result = greet();
  console.log(result);  // undefined
  ```

  * **조기 반환(Early Return) 패턴**
  ```JavaScript
  // 나쁜 예
  function processUser(user) {
      if (user) {
          if (user.age >= 18) {
              if (user.verified) {
                  // 실제 로직
                  return '처리 완료';
              } else {
                  return '인증 필요';
              }
          } else {
              return '미성년자';
          }
      } else {
          return '사용자 없음';
      }
  }

  // 좋은 예(조기 반환)
  function processUser(user) {
      if (!user) return '사용자 없음';
      if (user.age < 18) return '미성년자';
      if (!user.verified) return '인증 필요';

      // 실제 로직
      return '처리 완료';
  }
  ```
</details>

# 2. 고급 함수 패턴(함수 표현식, IIFE)

<details>
  <summary>(1) 이름 있는 함수 표현식(Named Function Expression)</summary>

  * **기본 형태**
  ```JavaScript
  const 변수명 = function 함수명(매개변수) {
      // ...
  };
  ```

  * **예제**
  ```JavaScript
  const factorial = function fact(n) {
      if (n <= 1) return 1;
      return n * fact(n - 1);  // 함수명으로 재귀 호출
  };

  console.log(factorial(5));  // 120

  // 함수명은 함수 내부에서만 사용 가능
  console.log(fact);  // ReferenceError
  ```

  * **언제 사용?**
  ```JavaScript
  // 1. 재귀 함수
  const countdown = function count(n) {
      if (n <= 0)  return;
      console.log(n);
      count(n - 1);  // 자기 자신 호출
  };

  countdown(5);  // 5, 4, 3, 2, 1

  // 2. 디버깅(스택 트레이스에 함수명 표시)
  const calculate = function calculateTotal(items) {
      // 에러 발생 시 'calculateTotal'이라고 표시됨
  };
  ```
</details>

<details>
  <summary>(2) 즉시 실행 함수(IIFE - Immediately Invoked Function Expression)</summary>

  * **기본 형태**
  ```JavaScript
  (function() {
      // 코드
  })();

  // 또는
  (function() {
      // 코드
  }());
  ```

  * **예제**
  ```JavaScript
  // 정의와 동시에 실행
  (function() {
      console.log('즉시 실행!');
  })();
  // 즉시 실행!

  // 매개변수 전달 가능
  (function(name) {
      console.log('안녕하세요, ' + name + '님!');
  })('홍길동');
  // 안녕하세요, 홍길동님!

  // 반환값 받기
  const result = (function() {
      result 10 + 20;
  })();

  console.log(result);  // 30
  ```

  * **왜 사용?**
  ```JavaScript
  // 1. 전역 스코프 오염 방지
  (function() {
      let privateVar = '비공개 변수';
      let count = 0;

      // 여기 변수들은 외부에서 접근 불가
  })();

  // console.log(privateVar); // ReferenceError

  // 2. 초기화 코드
  (function() {
        // 앱 초기 설정
        console.log('앱 초기화 중...');
        const config = loadConfig();
        initializeApp(config);
        console.log('초기화 완료!');
  })();

  // 3. 모듈 패턴 (ES6 이전)
  const counter = (function() {
      let count = 0; // private 변수

      return {
          increment: function() {
              count++;
              return count;
          },
          decrement: function() {
              count--;
              return count;
          },
          getCount: function() {
              return count;
          }
      };
  })();

  console.log(counter.increment());  // 1
  console.log(counter.increment());  // 2
  console.log(counter.getCount());   // 2
  // console.log(counter.count);     // undefined(접근 불가)
  ```

  * **현대적 대안**
  ```JavaScript
  // IIFE 대신 블록 사용(ES6+)
  {
      let privateVar = '비공개';
      const data = getData();
      processData(data);
  }

  // IIFE 대신 모듈 사용(ES6+)
  // counter.js
  let count = 0;

  export function increment() {
      count++;
      return count;
  }
  ```
</details>

# 3. 값으로서 함수⭐️

<details>
  <summary>(1) 일급 객체(First-class Object)</summary>

  * **자바스크립트에서 함수는 일급 객체**
  ```JavaScript
  일급 객체의 조건:
  1. 변수에 할당 가능
  2. 다른 함수의 인자로 전달 가능
  3. 다른 함수의 반환값으로 사용 가능
  ```
</details>

<details>
  <summary>(2) 변수에 할당</summary>

  ```JavaScript
  // 함수를 변수에 저장
  const greet = function() {
      console.log('안녕하세요!');
  };

  // 변수처럼 전달
  const sayHello = greet;
  sayHello();  // 안녕하세요!

  // 객체의 속성으로
  const person = {
      name: '홍길동',
      greet: function() {
          console.log('안녕하세요!');
      }
  };

  person.greet();  // 안녕하세요!!

  // 배열의 요소로
  const functions = [
      function() { return 1; },
      function() { return 2; },
      function() { return 3; }
  ];

  console.log(function[0]());  // 1
  console.log(function[1]());  // 2
  ```
</details>

<details>
  <summary>(3) 함수를 인자로 전달(콜백 함수)</summary>

  ```JavaScript
  // 기본 예제
  function runTwice(callback) {
      callback();
      callback();
  }

  runTwice(function() {
      console.log('실행!');
  });
  // 실행!
  // 실행!

  // 실용적인 예제
  function forEach(array, callback) {
      for (let i = 0; i < array.length; i++) {
          callback(array[i], i);
      }
  }

  forEach([1, 2, 3], function(item, index) {
      console.log(`${index}: ${item}`);
  });
  // 0: 1
  // 1: 2
  // 2: 3

  // 이벤트 핸들러
  button.addEventListener('click', function() {
      console.log('버튼 클릭!');
  });

  // setTimeout
  setTimeout(function() {
      console.log('3초 후 실행');
  }, 3000);
  ```
</details>

<details>
  <summary>(4) 함수를 반환값으로 사용</summary>

  ```JavaScript
  // 기본 예제
  function createGreeting(greeting) {
      return function(name) {
          return greeting + ', ' + name + '!';
      };
  }

  const sayHello = createGreeting('Hello');
  const sayHi = createGreeting('Hi');

  console.log(sayHello('John'));  // Hello, John!
  console.log(sayHi('Jane'));     // Hi, Jane!

  // 설정 함수 만들기
  function createMultiplier(multiplier) {
      return function(number) {
          return number * multiplier;
      };
  }

  const double = createMultiplier(2);
  const triple = createMultiplier(3);
  const quadruple = createMultiplier(4);

  console.log(double(5));     // 10
  console.log(triple(5));     // 15
  console.log(quadruple(5));  // 20

  // 조건부 함수 반환
  function getOperator(operator) {
      if (operator === '+') {
          return function(a, b) { return a + b; };
      } else if (operator === '-') {
          return function(a, b) { return a - b; };
      } else if (operator === '*') {
          return function(a, b) { return a * b; };
      } else if (operator === '/') {
          return function(a, b) { return a / b; };
      }
  }

  const add = getOperator('+');
  const multiply = getOperator('*');

  console.log(add(10, 5));       // 15
  console.log(multiply(10, 5));  // 50
  ```  
</details>

<details>
  <summary>(5) 고차 함수(Higher-order Function)</summary>

  * **정의**
    * 고차 함수: 함수를 인자로 받거나 함수를 반환하는 함수
    
  * **예제**
  ```JavaScript
  // 1. 함수를 인자로 받는 고차 함수
  function repeat(n, action) {
      for (let i = 0; i < n; i++) {
          action(i);
      }
  }

  repeat(3, function(i) {
      console.log(`${i}번째 실행`);
  });
  // 0번째 실행
  // 1번째 실행
  // 2번째 실행

  // 2. 함수를 반환하는 고차 함수
  function greaterThan(n) {
      return function(m) {
          return m > n;
      };
  }

  const greaterThan10 = greaterThan(10);
  console.log(greaterThan10(15));    // true
  console.log(greaterThan10(5));     // false

  // 3. 둘 다 하는 고차 함수
  function unless(test, then) {
      if (!test) then();
  }

  repeat(5, function(i) {
      unless(i % 2 === 0, function() {
          console.log(i + '는 홀수');
      });
  });
  // 1는 홀수
  // 3는 홀수
  ```
</details>
