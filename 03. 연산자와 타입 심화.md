# 1. 데이터 타입의 특징(원시 타입 vs 참조 타입)

<details>
  <summary>(1) 원시 타입(Primitive Types)</summary>
  
  * **종류**
  ```JavaScript
  // 5가지 원시 타입
  let num = 10;                // Number
  let str = 'Hello';           // String
  let bool = true;             // Boolean
  let nothing = null;          // null
  let notDefined = undefined;  // undefined
  ```

  * **특징: 값 자체가 저장됨**
  ```JavaScript
  let a = 10;
  let b = a; // 값이 복사됨

  b = 20;    // b만 변경

  console.log(a); // 10 (변하지 않음)
  console.log(b); // 20
  ```
  ```
  메모리 구조 :

  a: [10]
  b: [10] → [20]

  각각 독립적인 공간에 저장
  ```
                                       
</details>

<details>
  <summary>(2) 참조 타입(Reference Types)</summary>
  
  * **종류**
  ```JavaScript
  // 객체, 배열, 함수
  let obj = { name: '홍길동' };
  let arr = [1, 2, 3];
  let func = function() { };
  ```

  * **특징: 참조(주소)가 저장됨**
  ```JavaScript
  let person1 = { name: '홍길동', age: 25 };
  let person2 = person1; // 참조가 복사됨

  person2.age = 30;  // person2를 수정

  console.log(person1.age); // 30(같이 변경됨)
  console.log(person2.age);
  ```
  ```
  메모리 구조:

  person1: [주소: 0x001]
  person2: [주소: 0x001] (같은 주소)

  0x001: { name: '홍길동', age: 25 → 30 }

  같은 객체를 가리킴
  ```
  
</details>

<details>
  <summary>(3) 예제로 이해하기</summary>

  * **원시 타입**
  ```JavaScript
  function changeValue(x) {
    x = 100;
    console.log('함수 안: ', x);
  }

  let num = 10;
  changeValue(num);
  console.log('함수 밖: ', num);

  // 출력:
  // 함수 안: 100
  // 함수 밖: 10(변하지 않음)
  ``` 
    
  * **참조 타입**
  ```JavaScript
  function changeObject(obj) {
    obj.name = '김철수';
    console.log('함수 안: ', obj.name);
  }

  let person = { name: '홍길동' };
  changeObject(person);
  console.log('함수 밖: ', person.name);

  // 출력:
  // 함수 안: 김철수
  // 함수 밖: 김철수(같이 변경됨)
  ```
</details>

<details>
  <summary>(4) 배열과 객체 복사하기</summary>
  
  * **얕은 복사(Shallow Copy)**
  ```JavaScript
  // Spread 연산자 사용
  let original = [1, 2, 3];
  let copy = [...original];

  copy[0] = 100;

  console.log(original); // [1, 2, 3]
  console.log(copy);     // [100, 2, 3]

  // 객체도 마찬가지
  let person1 = { name: '홍길동', age: 25 };
  let person2 = { ...person1 };

  person2.age = 30;

  console.log(person1.age); // 25
  console.log(person2.age); // 30
  ```

  * **주의: 중첩된 객체는 여전히 참조**
  ```JavaScript
  let person1 = {
    name: '홍길동',
    address: {
        city: '서울'
    }
  };

  let person2 = { ...person1 };

  person2.address.city = '부산';

  console.log(person1.address.city); // 부산(같이 변경됨)
  console.log(person2.address.city); // 부산
  ```
  
  * **깊은 복사(Deep Copy)**
  ```JavaScript
  // structuredClone 사용(권장)
  let person1 = {
      name: '홍길동',
      address: {
          city: '서울'
      }
  };

  let person2 = structuredClone(person1);
  person2.address.city = '부산';

  console.log(person1.address.city); // 서울(원본 안전)
  console.log(person2.address.city); // 부산
  ``` 
</details>

<details>
  <summary>(5) 비교 연산</summary>

  * **원시 타입 비교**
  ```JavaScript
  let a = 10;
  let b = 10;
  console.log(a === b); // true(값이 같음)

  let str1 = 'hello';
  let str2 = 'hello';
  console.log(str1 === str2); // true
  ```

  * **참조 타입 비교**
  ```JavaScript
  let arr1 = [1, 2, 3];
  let arr2 = [1, 2, 3];
  console.log(arr1 === arr2); // false (다른 객체)

  let obj1 = { name: '홍길동' };
  let obj2 = { name: '홍길동' };
  console.log(obj1 === obj2); // false

  // 같은 참조만 true
  let arr3 = arr1;
  console.log(arr1 === arr3); // true (같은 객체)
  ```
</details>

# 3. Truthy & Falsy★★★

<details>
  <summary>(1) Falsy 값(거짓 같은 값)</summary>

  * **아래 8가지 빼고 다 truthy**
  ```JavaScript
  false, 0, -0, 0n, '', null, undefined, NaN
  ```
</details>

<details>
  <summary>(2) Truthy 값(참 같은 값)</summary>

  ```JavaScript
  true, 0을 제외한 숫자, 'hello', '0'(문자열 '0'은 Truthy), ' '(띄어쓰기), [], {}, function(){}
  ```
</details>

<details>
  <summary>(3) 활용</summary>

  * **값 존재 여부 체크**
  ```JavaScript
  let userName = '';

  if (userName) {
      console.log('이름: ', userName);
  }  else {
      console.log('이름이 없습니다.');
  }
  // 이름이 없습니다.

  userName = '홍길동';

  if (userName) {
      console.log('이름: ', userName);
  }
  // 이름: 홍길동
  ```

  * **기본값 설정**
  ```JavaScript
  function greet(name) {
      // name이 Falsy면 기본값 사용
      name = name || '게스트';
      console.log('안녕하세요, ' + name + '님!');
  }

  greet('홍길동'); // 안녕하세요, 홍길동님!
  greet('');      // 안녕하세요, 게스트님!
  greet();        // 안녕하세요, 게스트님!
  ```

  * **배열이 비어있는지 체크**
  ```JavaScript
  let items = [];

  // 잘못된 방법 (배열은 Truthy)
  if (items) {
      console.log('아이템 있음');
  }
  // 아이템 있음(빈 배열인데도!)

  // 올바른 방법
  if (items.length > 0) {
      console.log('아이템 있음');
  } else {
      console.log('아이템 없음');
  }
  // 아이템 없음
  ```

</details>

<details>
  <summary>(4) 명시적 변환</summary>

  * **Boolean() 함수**
  ```JavaScript
  // Falsy → false
  console.log(Boolean(0));          // false  
  console.log(Boolean(''));         // false
  console.log(Boolean(null));       // false
  console.log(Boolean(undefined));  // false

  // Truthy → true
  console.log(Boolean(1));          // true
  console.log(Boolean('hello'));    // true
  console.log(Boolean([]));         // true
  console.log(Boolean({}));         // true
  ```

  * **!!(이중 부정)**
  ```JavaScript
  // Falsy → false
  console.log(!!0);        // false
  console.log(!!'');       // false

  // Truthy → true
  console.log(!!1);        // true
  console.log(!!'hello');  // true
  console.log(!![]);       // true
  ```

</details>
